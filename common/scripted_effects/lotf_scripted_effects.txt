pismark_capitulation_event_scripted_effect = {
	if = {
		limit = {
			ROOT = MIN
		}
		country_event = bismark_capitualtion.1
		set_global_flag = bismark_dead
	}
	else_if = {
		limit = {
			ROOT = MIL
		}
		country_event = bismark_capitualtion.2
		set_global_flag = bismark_dead
	}
	else_if = {
		limit = {
			ROOT = ABS
		}
		country_event = bismark_capitualtion.72
		set_global_flag = bismark_dead
	}
}

ief_improve_professional_army = {
	if = {
		limit = {
			has_idea = IEF_professional_army_01
		}
		swap_ideas = {
			remove_idea = IEF_professional_army_01
			add_idea = IEF_professional_army_02
		}
	}
	else_if = {
		limit = {
			has_idea = IEF_professional_army_02
		}
		swap_ideas = {
			remove_idea = IEF_professional_army_02
			add_idea = IEF_professional_army_03
		}
	}
}
